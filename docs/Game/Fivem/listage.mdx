# Comment listée sont serveur Fivem

Pour listée sont serveur FiveM il faut aller sur le fichier server.cfg et ajouter les ligne suivantes:

sv_endpointPrivacy true
:::

:::tip L'ip du serveur peut être l'ip du sous domaine (Gamex.hebergtonserv.com:leport par exemple)

:::

set sv_endpoints "ipduserveur:port" 
### listing view fivem ###
set sv_useDirectListing true # listing view fivem
set sv_listingIPOverride "gamex.hebergtonserv.com:votreport"  # (Cela doit être pareil que votre sv_endpoints) 
sets sv_projectName "[FR]^1 LSDream FA:nazar_amulet:・^7discord.gg/^147ACRnWSd3" # Set un nom de projet pour votre projet
sets sv_projectDesc "Old Jersey " # Set une description pour le projet
set onesync on # on(infinity= +64slot )/legacy(legacy= -64slot) # Permet l'activation du onesync
set steam_webApiKey none # Clé API STEAM
set sv_licenseKey VOTRE_CLE # Identifiant de votre clé FiveM
sv_maxclients 64 # Nombre maximum de slots sur votre serveur FiveM
